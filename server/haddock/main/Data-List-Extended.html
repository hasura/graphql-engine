longestCommonPrefix :: Eq a => [[a]] -> [a] Source #

Returns the longest prefix common to all given lists. Returns an empty list on an empty list.

>>> longestCommonPrefix ["abcd", "abce", "abgh"]
"ab"

>>> longestCommonPrefix []
[]
